#Task 5
mkdir ~/localrepo
wget https://github.com/fedos3d/itmo_linux/raw/master/checkinstall-1.6.2-3.el6.1.x86_64.rpm

# let's create file localrepo.repo in /etc/yum.repos.d/

yum repolist > ~/task6.log

#Task 7
